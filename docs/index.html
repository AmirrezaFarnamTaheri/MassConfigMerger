Simple to Use

# Fetch and merge configurations
configstream full

# Fetch from last 12 hours
configstream fetch --hours 12

# Test and merge with filtering
configstream merge --include-protocols VMESS,VLESS --top-n 100

# Retest existing subscription
configstream retest output/vpn_subscription_raw.txt
# Advanced filtering and options
configstream full \
    --hours 24 \
    --fetch-protocols vmess,vless,trojan \
    --include-country US,GB,JP \
    --exclude-pattern "free|trial" \
    --top-n 200 \
    --max-ping 500 \
    --concurrent-limit 50 \
    --output-surge output/surge.conf

# Custom configuration
configstream merge \
    --config custom_config.yaml \
    --output-dir /path/to/output \
    --no-sort \
    --no-base64
# Run with Docker
docker run -v $(pwd)/output:/app/output \
    configstream/configstream \
    configstream full

# Docker Compose
docker-compose up -d

# Access web interface
docker-compose up web
# Open http://localhost:8080
